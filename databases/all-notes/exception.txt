delimiter //

create procedure divide(in a int, in b int)
begin
declare vans int;
declare mydividebyzero condition for SQLSTATE '45000';
declare exit handler for mydividebyzero resignal set MESSAGE_TEXT = "Can't divide by zero";

if b = 0 then signal mydividebyzero;
else set vans = a/b;
end if;
select vans;
end //
