create procedure displaySeq2(in n int, out patt varchar(300))
begin 
declare str varchar(300) default "";
declare cnt int default 1;
while cnt <= n do
  set str = concat(str, cnt, ',');
  set cnt  = cnt +1;
end while;

set str = substr(str, 1, length(str)-1);
select str;
set patt = str;
end//
